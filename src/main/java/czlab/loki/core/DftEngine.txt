package czlab.loki.core;

/**
 * @author Kenneth Leung
 */
public abstract class DftEngine implements Engine {

  protected Object _container;
  protected Object _state;

  /**
   */
  public abstract void onInit(Object arg);

  @Override
  public void init(Object state) {
    onInit(_state=state);
  }

  @Override
  public Object restart(Object arg0) {
    return null;
  }

  @Override
  public Object start(Object arg0) {
    return null;
  }

  @Override
  public void stop() {

  }

  @Override
  public void startRound(Object arg) {
  }

  @Override
  public void endRound(Object any) {
  }

  @Override
  public void update(Object event) {
  }

  @Override
  public Object ready(Object container) {
    _container= container;
    return this;
  }

  @Override
  public Object state() {
    return _state;
  }

  @Override
  public Object container() {
    return _container;
  }

}
